query TournamentPhaseGroupQuery($id: ID!, $videogameId: ID) {
  phaseGroup(id: $id) {
    seedMap
    seeds(query: {page: 1, perPage: 500}) {
      nodes {
        seedNum
        groupSeedNum
        progressionSource {
          originPhaseGroup {
            id
          }
        }
        entrant {
          id
          name
          participants {
            player {
              id
              gamerTag
              prefix
              name
            }
            user {
              id
              prefix
              name
              genderPronoun
              location {
                country
                state
                city
              }
              authorizations(types: [TWITTER]) {
                externalUsername
              }
              images(type: "profile") {
                url
              }
            }
          }
        }
      }
    }
    sets(page: 1, perPage: 9999, filters: {showByes: true, hideEmpty: false}, sortType: CALL_ORDER){
      nodes {
        id
        round
        entrant1Id
        entrant2Id
        winnerId
        entrant1Score
        entrant2Score
        state
        slots {
          prereqType
        }
      }
    }
    progressionsOut {
      id
    }
  }
}
